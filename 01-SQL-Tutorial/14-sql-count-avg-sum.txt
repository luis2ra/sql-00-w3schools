SQL COUNT(), AVG() and SUM() Functions

# COUNT() Syntax
SELECT COUNT(column_name)
FROM table_name
WHERE condition;


# AVG() Syntax
SELECT AVG(column_name)
FROM table_name
WHERE condition;


# SUM() Syntax
SELECT SUM(column_name)
FROM table_name
WHERE condition;
